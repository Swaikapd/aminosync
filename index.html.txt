<script>
function calculateEAA() {
    let weight = parseFloat(document.getElementById('weight').value);
    let unit = document.getElementById('weightUnit').value;
    let goal = document.getElementById('goal').value;
    let intensity = document.getElementById('intensity').value;
    let stomach = document.querySelector('input[name="stomach"]:checked').value;

    if (isNaN(weight) || weight <= 0) { alert("Masukkan berat badan."); return; }

    let weightKg = unit === 'lbs' ? weight * 0.453592 : weight;
    let multiplier = 0.15; 
    
    if (goal === 'hypertrophy') multiplier += 0.03;
    if (goal === 'fatloss') multiplier += 0.05; 
    if (intensity === 'light') multiplier -= 0.03;
    if (intensity === 'intense') multiplier += 0.04;

    let totalEAA = Math.min(Math.max(weightKg * multiplier, 6), 25);
    document.getElementById('totalDose').innerText = totalEAA.toFixed(1) + " g";

    let html = "";
    if (stomach === 'fasted') {
        html += `<tr><td><strong>Pre-Workout</strong><br><small>15m sebelum latihan</small></td><td>${(totalEAA * 0.6).toFixed(1)} g</td><td><span class="badge badge-rec">Kritis (Mencegah Katabolik)</span></td></tr>`;
        html += `<tr><td><strong>Intra-Workout</strong><br><small>Sip pelan saat latihan</small></td><td>${(totalEAA * 0.4).toFixed(1)} g</td><td><span class="badge badge-opt">Pendukung Energi</span></td></tr>`;
    } else {
        html += `<tr><td><strong>Pre-Workout</strong><br><small>15m sebelum latihan</small></td><td>0 g</td><td><span class="badge badge-opt">Tidak wajib (Sudah ada asam amino darah)</span></td></tr>`;
        html += `<tr><td><strong>Intra-Workout</strong><br><small>Sip pelan saat latihan</small></td><td>${(totalEAA * 0.5).toFixed(1)} g</td><td><span class="badge badge-rec">Sangat Direkomendasikan</span></td></tr>`;
        html += `<tr><td><strong>Post-Workout</strong><br><small>Jika jeda makan besar lambat</small></td><td>${(totalEAA * 0.5).toFixed(1)} g</td><td><span class="badge badge-opt">Opsional</span></td></tr>`;
    }
    document.getElementById('timingBody').innerHTML = html;
    document.getElementById('results').classList.add('active');
}
</script>
